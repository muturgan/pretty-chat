(
    `id` INT PRIMARY KEY AUTO_INCREMENT UNIQUE,
    `date` BIGINT UNSIGNED,
    `room` VARCHAR(20) DEFAULT "public",
    `text` TEXT,
    `author_id` INT REFERENCES `users` (id) ON DELETE NO ACTION
);
